# Create a 4x4 matrix filled with 0's
matrix_zeros <- matrix(0, nrow = 4, ncol = 4)

# Add a tie between node 1 and node 3 by placing a 1
matrix_zeros[1, 3] <- 1  # From node 1 to node 3


# Print the matrix
print(matrix_zeros)
#>      [,1] [,2] [,3] [,4]
#> [1,]    0    0    1    0
#> [2,]    0    0    0    0
#> [3,]    0    0    0    0
#> [4,]    0    0    0    0
